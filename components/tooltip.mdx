The Tooltip component displays contextual information when users interact with chart elements, providing detailed data values and descriptions.

## Purpose

Based on the source implementation in `~/workspace/source/src/component/tooltip/TooltipModel.ts`, the tooltip component:

- Displays data details on mouse hover or click events
- Supports multiple trigger modes (item-based or axis-based)
- Can render in HTML or canvas/SVG (richText) mode
- Provides customizable formatting and styling
- Includes axis pointer indicators for better data reference

## Basic Usage

```javascript
option = {
  tooltip: {
    trigger: 'axis',
    axisPointer: {
      type: 'line'
    }
  },
  xAxis: { type: 'category', data: ['Mon', 'Tue', 'Wed'] },
  yAxis: { type: 'value' },
  series: [{
    type: 'line',
    data: [120, 200, 150]
  }]
};
```

## Item Trigger Tooltip

```javascript
option = {
  tooltip: {
    trigger: 'item',
    formatter: '{b}: {c} ({d}%)'
  },
  series: [{
    type: 'pie',
    data: [
      { value: 335, name: 'Direct' },
      { value: 310, name: 'Email' },
      { value: 234, name: 'Ads' }
    ]
  }]
};
```

## Custom Tooltip Content

```javascript
option = {
  tooltip: {
    trigger: 'axis',
    formatter: function(params) {
      let result = params[0].axisValue + '<br/>';
      params.forEach(function(item) {
        result += item.marker + ' ' + item.seriesName + ': ' + item.value + '<br/>';
      });
      return result;
    },
    backgroundColor: 'rgba(50,50,50,0.7)',
    borderColor: '#333',
    borderWidth: 1,
    textStyle: {
      color: '#fff'
    }
  }
};
```

## Axis Pointer with Cross Style

```javascript
option = {
  tooltip: {
    trigger: 'axis',
    axisPointer: {
      type: 'cross',
      crossStyle: {
        color: '#999',
        width: 1,
        type: 'dashed'

    }
  },
  xAxis: { type: 'category', data: ['A', 'B', 'C'] },
  yAxis: { type: 'value' },
  series: [{ type: 'line', data: [10, 20, 30] }]
};
```

## Configuration Options

### Core Properties

<ParamField path="show" type="boolean" default="true">
  Whether to display the tooltip component. From `TooltipModel.ts:99`.
</ParamField>

<ParamField path="trigger" type="'item' | 'axis' | 'none'" default="'item'">
  Trigger type for tooltip:
  - `'item'`: Triggered by data items (default)
  - `'axis'`: Triggered by axes, shows all series at the same axis value
  - `'none'`: No automatic trigger
  
  Implementation in `TooltipModel.ts:106`.
</ParamField>

<ParamField path="triggerOn" type="string" default="'mousemove|click'">
  Conditions to trigger tooltip. Can be:
  - `'mousemove'`: Trigger on mouse movement
  - `'click'`: Trigger on click
  - `'mousemove|click'`: Trigger on either event
  - `'none'`: Manual trigger only
  
  From `TooltipModel.ts:109`.
</ParamField>

<ParamField path="showContent" type="boolean" default="true">
  Whether to show tooltip content. Set to false to only show axis pointer without content box.
  
  See `TooltipModel.ts:102`.
</ParamField>

### Rendering Mode

<ParamField path="renderMode" type="'auto' | 'html' | 'richText'" default="'auto'">
  Rendering mode for tooltip:
  - `'auto'`: Use HTML by default, falls back to richText if document is unavailable
  - `'html'`: Render tooltip as HTML DOM element
  - `'richText'`: Render using canvas/SVG graphics
  
  Implementation in `TooltipModel.ts:113`.
</ParamField>

<ParamField path="appendTo" type="function | string | HTMLElement">
  Specify where to append the tooltip DOM element. Only works when `renderMode` is 'html'.
  
  ```javascript
  appendTo: document.body
  // or
  appendTo: 'body'
  // or
  appendTo: (chartContainer) => chartContainer.parentNode
  ```
  
  From `TooltipModel.ts:72`.
</ParamField>

<ParamField path="className" type="string">
  CSS class name for the tooltip element. Only available when `renderMode` is 'html'.
  
  See `TooltipModel.ts:78`.
</ParamField>

<ParamField path="transitionDuration" type="number" default="0.4">

<ParamField path="alwaysShowContent" type="boolean" default="false">

  From `TooltipModel.ts:111`.

<ParamField path="enterable" type="boolean" default="false">

### Positioning

<ParamField path="confine" type="boolean" default="null">
  Whether to confine tooltip within the chart area:

  - For `renderMode: 'html'`: defaults to `false` (for backward compatibility)
  
  From `TooltipModel.ts:118`.
</ParamField>

### Content Formatting

<ParamField path="formatter" type="string | function">
  Tooltip content formatter. Can be:
  
  **String template** with variables:
  - `{a}`: series name
  - `{b}`: data name / category
  - `{c}`: data value
  - `{d}`: percentage (for pie charts)
  
  ```javascript
  formatter: '{b}: {c}'
  ```
  
  **Function** for custom formatting:
  ```javascript
  formatter: function(params) {
    return params.name + ': ' + params.value;
  }
  ```
</ParamField>

<ParamField path="order" type="TooltipOrderMode">
  Order of items in tooltip when `trigger: 'axis'`. Controls how multiple series are sorted.
</ParamField>

## Advanced Features

### Multi-Series Default Border

<ParamField path="defaultBorderColor" type="string">
  Default border color when there are multiple series. Useful for maintaining visual consistency.
  
  From `TooltipModel.ts:145`.
</ParamField>

### Rich Text Rendering

When using `renderMode: 'richText'`, tooltips are rendered using canvas/SVG, which provides:
- Better performance with large datasets
- No DOM manipulation overhead
- Consistent rendering across environments
- Automatic confinement to chart bounds

## Example: Complete Tooltip Configuration

```javascript
option = {
  tooltip: {
    trigger: 'axis',
    showContent: true,
    renderMode: 'html',
    triggerOn: 'mousemove',
    showDelay: 0,
    hideDelay: 100,
    transitionDuration: 0.4,
    enterable: false,
    confine: false,
    backgroundColor: 'rgba(50, 50, 50, 0.9)',
    borderColor: '#333',
    borderWidth: 1,
    borderRadius: 4,
    padding: [10, 15],
    textStyle: {
      color: '#fff',
      fontSize: 14
    },
    formatter: function(params) {
      let html = '<div style="font-weight: bold;">' + params[0].axisValue + '</div>';
      params.forEach(function(item) {
        html += '<div>' + item.marker + ' ' + item.seriesName + ': ' + item.value + '</div>';
      });
      return html;
    },
    axisPointer: {
      type: 'cross',
      axis: 'auto',
      animation: true,
      animationDurationUpdate: 200,
      crossStyle: {
        color: '#999',
        width: 1,
        type: 'dashed'
      }
    }
  },
  xAxis: { type: 'category', data: ['Mon', 'Tue', 'Wed', 'Thu', 'Fri'] },
  yAxis: { type: 'value' },
  series: [
    { name: 'Sales', type: 'line', data: [120, 200, 150, 80, 70] },
    { name: 'Revenue', type: 'line', data: [100, 180, 130, 90, 60] }
  ]
};
```
